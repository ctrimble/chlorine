16  package com.github.ctrimble.chlorine.classloader;

18  import java.net.URL;
19  import java.net.URLClassLoader;
20  import java.net.URLStreamHandlerFactory;
21  import java.nio.ByteBuffer;
22  import java.security.CodeSource;
23  import java.security.ProtectionDomain;
24  
25  import com.github.ctrimble.chlorine.asm.ClassByteLoader;
26  import com.github.ctrimble.chlorine.asm.ClassTransformer;

28  /**
29   * Chlorine's replacement for java.net.URLClassLoader.  In cases where a URLClassLoader is instantiated,
30   * this class will be used instead.  In cases where a non java/javax class extends URLClassLoader, this
31   * class will be injected into the class hierarchy.
32   * 
33   * @author Christian Trimble
34   */
35  public class ChlorineURLClassLoader extends URLClassLoader implements ChlorineSecureClassDefiner {
36  	private ClassByteLoader loader;
37  
38  	public ChlorineURLClassLoader(URL[] urls) {
39  		super(urls);
40  		loader = new ClassByteLoader(this);
41  	}
42  
43  	public ChlorineURLClassLoader(URL[] urls, ClassLoader parent) {
44  		super(urls, parent);
45  		loader = new ClassByteLoader(this);
46  	}
47  
48  	public ChlorineURLClassLoader(URL[] urls, ClassLoader parent, URLStreamHandlerFactory factory) {
49  		super(urls, parent, factory);
50  		loader = new ClassByteLoader(this);
51  	}
52  	
53  	@SuppressWarnings("deprecation")
54  	public Class<?> chlorineDefineClass(byte[] b, int off, int len)
55  			throws ClassFormatError {
56  		byte[] newBytes = ClassTransformer.transform(b, off, len, loader);
57  		return defineClass(newBytes, 0, newBytes.length);
58  	}
59  
60  	public Class<?> chlorineDefineClass(String name, byte[] b, int off, int len)
61  			throws ClassFormatError {
62  		byte[] newBytes = ClassTransformer.transform(b, off, len, loader);
63  		return defineClass(name, newBytes, 0, newBytes.length);
64  	}
65  
66  	public Class<?> chlorineDefineClass(String name, byte[] b, int off, int len,
67  			ProtectionDomain protectionDomain) throws ClassFormatError {
68  		byte[] newBytes = ClassTransformer.transform(b, off, len, loader);
69  		return defineClass(name,  newBytes, 0, newBytes.length, protectionDomain);
70  	}
71  
72  	public Class<?> chlorineDefineClass(String name, ByteBuffer b,
73  			ProtectionDomain protectionDomain) throws ClassFormatError {
74  		byte[] newBytes = new byte[b.remaining()];
75  	    b.get(newBytes);
76  		return defineClass(name, newBytes, 0, newBytes.length, protectionDomain);
77  	}
78  
79  	public Class<?> chlorineDefineClass(String name, byte[] b, int off, int len, CodeSource cs) {
80  		byte[] newBytes = ClassTransformer.transform(b, off, len, loader);
81  		return defineClass(name, newBytes, 0, newBytes.length, cs);
82  	}
83  
84  	public Class<?> chlorineDefineClass(String name, ByteBuffer b, CodeSource cs) {
85  		byte[] newBytes = new byte[b.remaining()];
86  	    b.get(newBytes);
87  		return defineClass(name, newBytes, 0, newBytes.length, cs);
88  	}
89  }
