29  /**
30   * A class visitor for ASM 4 that injects Chlorine class loaders into java class files.
31   * 
32   * @author Christian Trimble
33   */
34  public class DefineClassAdapter extends ClassVisitor {
35      private static String CHLORINE_CLASSLOADER_PACKAGE = "com/github/ctrimble/chlorine/classloader/";
36      private static String CHLORINE_DEFINE_CLASS_NAME = "chlorineDefineClass";
37      private static String CLASS_LOADER_NAME = "java/lang/ClassLoader";
38      private static String SECURE_CLASS_LOADER_NAME = "java/security/SecureClassLoader";
39      private static String URL_CLASS_LOADER_NAME = "java/net/URLClassLoader";
40      private static String CHLORINE_SECURE_CLASS_LOADER_NAME = "com/github/ctrimble/chlorine/classloader/ChlorineSecureClassLoader";
41      private static String CHLORINE_CLASS_LOADER_NAME = "com/github/ctrimble/chlorine/classloader/ChlorineClassLoader";
42      private static String CHLORINE_URL_CLASS_LOADER_NAME = "com/github/ctrimble/chlorine/classloader/ChlorineURLClassLoader";
43      private static Set<String> DEFINE_CLASS_DESCS = new HashSet<String>();
44      private static Set<String> DEFINE_CLASS_SECURE_DESCS = new HashSet<String>();
45      static {
46      	DEFINE_CLASS_DESCS.add("([BII)Ljava/lang/Class;");
47      	DEFINE_CLASS_DESCS.add("(Ljava/lang/String;[BII)Ljava/lang/Class;");
48      	DEFINE_CLASS_DESCS.add("(Ljava/lang/String;[BIILjava/security/ProtectionDomain;)Ljava/lang/Class;");
49      	DEFINE_CLASS_DESCS.add("(Ljava/lang/String;Ljava/nio/ByteBuffer;Ljava/security/ProtectionDomain;)Ljava/lang/Class;");
50      	// TODO: get the SecureClassLoader defineClass descs in here.
51      	//DEFINE_CLASS_SECURE_DESCS
52      	//DEFINE_CLASS_SECURE_DESCS
53      }

55  	private ClassReader source;
56      private ClassByteLoader loader;
57  	public DefineClassAdapter(ClassReader source, ClassByteLoader loader, ClassVisitor next) {
58  		super(ASM4, next);
59  		this.source = source;
60  		this.loader = loader;
61  	}
62  	
63      private Boolean isClassLoader = null;
64      private Boolean isSecureClassLoader = null;
65      private Boolean isURLClassLoader = null;
66      
67      private Boolean isClassLoader() {
68      	if( isClassLoader == null ) isClassLoader = CLASS_LOADER_NAME.equals(source.getClassName()) || hasAncestor( source, loader, CLASS_LOADER_NAME);
69      	return isClassLoader;
70      }
71      
72      private Boolean isSecureClassLoader() {
73      	if( isSecureClassLoader == null ) isSecureClassLoader = SECURE_CLASS_LOADER_NAME.equals(source.getClassName()) || hasAncestor( source, loader, SECURE_CLASS_LOADER_NAME);
74      	return isSecureClassLoader;
75      }
76      
77      public Boolean isURLClassLoader() {
78      	if( isURLClassLoader == null ) isURLClassLoader = URL_CLASS_LOADER_NAME.equals(source.getClassName()) || hasAncestor( source, loader, URL_CLASS_LOADER_NAME);
79      	return isURLClassLoader;    	
80      }
81      
82      private static boolean hasAncestor( ClassReader reader, ClassByteLoader loader, String internalName ) {
83      	String superName = reader.getSuperName();
84      	if( superName.equals(internalName) ) return true;
85      	else if( superName.equals("java/lang/Object") ) return false;
86      	else return hasAncestor( new ClassReader(loader.getBytes(superName)), loader, internalName);
87      }
88  	
89  	@Override
90  	public MethodVisitor visitMethod(int access, String name, String desc, String signature, String[] exceptions) {
91        if(name.startsWith(CHLORINE_CLASSLOADER_PACKAGE)) return super.visitMethod(access, name, desc, signature, exceptions);
92        else return new DefineClassVisitor(super.visitMethod(access, name, desc, signature, exceptions));
93  	}

95  	/**
96  	 * An ASM 4 class visitor that replaces instantiations of java class loaders with Chlorine class loaders.
97  	 * 
98  	 * @author Christian Trimble
99  	 */
100 	public class DefineClassVisitor
101 	    extends MethodVisitor
102 	{
103 		public DefineClassVisitor(MethodVisitor mv) {
104 			super(ASM4, mv);
105 		}
106 		
107 		@Override
108 		public void visitMethodInsn(int opcode, String owner, String name, String desc) {
109 			if( opcode == INVOKEVIRTUAL && "defineClass".equals(name) &&
110 		        ( (DEFINE_CLASS_DESCS.contains(desc) && isClassLoader() ) ||
111 		          (DEFINE_CLASS_SECURE_DESCS.contains(desc) && isSecureClassLoader() ) ) ) {
112 				if( isURLClassLoader() )
113 					mv.visitMethodInsn(opcode, CHLORINE_URL_CLASS_LOADER_NAME, CHLORINE_DEFINE_CLASS_NAME, desc);
114 				else if( isSecureClassLoader() )
115 				    mv.visitMethodInsn(opcode, CHLORINE_SECURE_CLASS_LOADER_NAME, CHLORINE_DEFINE_CLASS_NAME, desc);
116 				else if( isClassLoader() ) {
117 					mv.visitMethodInsn(opcode, CHLORINE_CLASS_LOADER_NAME, CHLORINE_DEFINE_CLASS_NAME, desc);
118 				}
119 			}
120 			else super.visitMethodInsn(opcode, owner, name, desc);
121 		}
